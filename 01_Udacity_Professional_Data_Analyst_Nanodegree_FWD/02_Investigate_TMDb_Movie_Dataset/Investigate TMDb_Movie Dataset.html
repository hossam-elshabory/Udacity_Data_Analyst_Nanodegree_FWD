cols = ["revenue", "budget"]
def remv_zeros(columns):
    """ Takes in columns names and replaces any zero values with NAN then drop them"""
    df[columns] = df[columns].replace(0, np.NAN)
    var1 = df.query(f"{columns[0]} == 0").value_counts()
    var2 = df.query(f"{columns[1]} == 0").value_counts()
    if 0 not in var1 and 0 not in var2:
        return f"{columns[0]} have no 0 values", f"{columns[1]} have no 0 values"
    df.dropna(inplace=True)


remv_zeros(cols)
